javascript:(function() {
  const s = document.createElement('script');
  s.src = 'https://raw.githubusercontent.com/username/repo/branch/config.js';
  s.onload = () => {
    showDropdown(window.bookmarkletMenuItems);
  };
  document.head.appendChild(s);

  function showDropdown(items) {
    const div = document.createElement('div');
    div.id = 'bm-dropdown';
    div.style = 'position:fixed;top:10px;right:10px;z-index:9999;background:white;border:1px solid #ccc;padding:10px;';
    const select = document.createElement('select');
    select.innerHTML = '<option value="">Select bookmarklet...</option>';
    items.forEach(item => {
      const opt = document.createElement('option');
      opt.value = item.url;
      opt.textContent = item.name;
      select.appendChild(opt);
    });
    select.onchange = () => {
      if (select.value) {
        const script = document.createElement('script');
        script.src = select.value;
        document.head.appendChild(script);
      }
      select.selectedIndex = 0;
    };
    div.appendChild(select);
    document.body.appendChild(div);
  }
})();
